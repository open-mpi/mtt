145  # scan the key-value pairs obtained from the configuration
146  # parser and compare them with the options defined for a
147  # given plugin. Generate an output dictionary that contains
148  # the updated set of option values, the default value for
149  # any option that wasn't included in the configuration file,
150  # and return an error status plus output identifying any
151  # keys in the configuration file that are not supported
152  # by the list of options
153  #
154  # @log [INPUT]
155  # - a dictionary that will return the status plus
156  # stderr containing strings identifying any
157  # provided keyvals that don't have a corresponding
158  # supported option
159  # @options [INPUT]
160  # - a dictionary of tuples, each consisting of three
161  # entries:
162  # (a) the default value
163  # (b) data type
164  # (c) a help-like description
165  # @keyvals [INPUT]
166  # - a dictionary of key-value pairs obtained from
167  # the configuration parser
168  # @target [OUTPUT]
169  # - the resulting dictionary of key-value pairs
170  def parseOptions(self, log, options, keyvals, target):
171  # parse the incoming keyvals dictionary against the source
172  # options. If a source option isn't provided, then
173  # copy it across to the target.
174  opts = list(options.keys())
175  kvkeys = list(keyvals.keys())
176  for opt in opts:
177  found = False
178  for kvkey in kvkeys:
179  if kvkey == opt:
180  # they provided us with an update, so
181  # pass this value into the target - expand
182  # any provided lists
183  if keyvals[kvkey] is None:
184  continue
185  st, outval = self.__convert_value(options[opt][0], keyvals[kvkey])
186  if 0 == st:
187  target[opt] = outval
188  else:
189  if len(keyvals[kvkey]) == 0:
190  # this indicates they do not want this option
191  found = True
192  break
193  if keyvals[kvkey][0][0] == "[":
194  # they provided a list - remove the brackets
195  val = keyvals[kvkey].replace('[','')
196  val = val.replace(']','')
197  # split the input to pickup sets of options
198  newvals = list(val)
199  # convert the values to specified type
200  i=0
201  for val in newvals:
202  st, newvals[i] = self.__convert_value(opt[0], val)
203  i = i + 1
204  target[opt] = newvals
205  else:
206  st, target[opt] = self.__convert_value(opt[0], keyvals[kvkey])
207  found = True
208  break
209  if not found:
210  # they didn't provide this one, so
211  # transfer only the value across
212  target[opt] = options[opt][0]
213  # add in any default settings that have not
214  # been overridden - anything set by this input
215  # stage will override the default
216  if self.defaults is not None:
217  keys = list(self.defaults.options.keys())
218  for key in keys:
219  if key not in target:
220  target[key] = self.defaults.options[key][0]
221 
222  # now go thru in the reverse direction to see
223  # if any keyvals they provided aren't supported
224  # as this would be an error
225  unsupported_options = []
226  for kvkey in kvkeys:
227  # ignore some standard keys
228  if kvkey in ['section', 'plugin']:
229  continue
230  try:
231  if target[kvkey] is not None:
232  pass
233  except KeyError:
234  # some always need to be passed
235  if kvkey in ['parent', 'asis']:
236  target[kvkey] = keyvals[kvkey]
237  else:
238  unsupported_options.append(kvkey)
239  if unsupported_options:
240  sys.exit("ERROR: Unsupported options for section [%s]: %s" % (log['section'], ",".join(unsupported_options)))
241 
242  log['status'] = 0
243  log['options'] = target
244  return

816  max2 = max2 + 4
